//il faut avoir 2 terminal
dans 1 terminal tu lances flask et dans l'autres tu fais  curl nanana

option de curl: -i : --include: inclus les entete HTTP dans la réponse
-H : header, entête
-X : type de requête (GET, POST, PUT,...)
-d : data, les données envoyé

#POST
curl -i -H "Content-Type: application/json" -X POST -d '{"nom":"hola"}' http://localhost:5000/quiz/api/v1.0/questionaires

#POST
curl -i -X DELETE http://localhost:5000/quiz/api/v1.0/questionaires/2

#Avec Question
#POST
curl -i -H "Content-Type: application/json" -X POST -d '{"enonce":"Quelle ?"}' http://localhost:5000/quiz/api/v1.0/questionnaires/1/questions
#powershell
Invoke-RestMethod -Method Post -Uri "http://localhost:5000/quiz/api/v1.0/questionaires/1/questions" -ContentType "application/json" -Body '{"enonce":"Quelle est la capitale de la France ?"}'

#DELETE
curl -i -X DELETE http://localhost:5000/quiz/api/v1.0/questionaires/1/questions/1

#powershell
Invoke-RestMethod -Method Delete -Uri "http://localhost:5000/quiz/api/v1.0/questionaires/1/questions/1"
